# -----------------------------------------------------------------------------
# BASH SAFETY FLAGS
# -----------------------------------------------------------------------------
# set -e  → exit immediately if any command returns a non-zero exit code
# set -u  → treat unset variables as errors (prevents typo-variable bugs)
# set -o pipefail → if any command in a pipeline fails, the whole pipeline fails
set -euo pipefail

# -----------------------------------------------------------------------------
# COLOR CODES
# -----------------------------------------------------------------------------
# ANSI escape sequences used to colorize terminal output.
# \033[ is the escape sequence prefix; the number selects the color/style.
on_yellow='\033[43m'  # Yellow background
on_green='\033[42m'   # Green background
b_yellow='\033[1;33m' # Bold Yellow text
b_black='\033[1;30m'  # Bold Black text
b_green='\033[1;32m'  # Bold Green text
green='\033[0;32m'    # Light Green text
red='\033[0;31m'      # Red text
nc='\033[0m'          # No Color — resets formatting back to terminal default

# Additional bold color aliases used in later sections
CYAN='\033[1;36m'
GREEN='\033[1;32m'
RED='\033[1;31m'
YELLOW='\033[1;33m'

# -----------------------------------------------------------------------------
# HELPER FUNCTIONS
# -----------------------------------------------------------------------------

# step() — clears the screen, redraws the banner, and prints a bold step header.
# Argument $1 is the step title text to display.
step() {
    clear          # clear the terminal screen
    show_banner    # redraw the ASCII art banner (defined below)
    echo ""
    echo -e "${CYAN}===========================================================  ${nc}"
    echo -e "${CYAN}  $1${nc}"
    echo -e "${CYAN}===========================================================  ${nc}"
    echo ""
}

# ok() — prints a green [OK] success message.
ok() {
    echo -e "${GREEN}  [OK] $1${nc}"
}

# info() — prints a yellow [INFO] informational message.
info() {
    echo -e "${YELLOW}  [INFO] $1${nc}"
}

# die() — prints a red [ERROR] message and exits the script with code 1.
die() {
    echo -e "${RED}  [ERROR] $1${nc}"
    exit 1
}

# progress() — displays a fake animated progress bar using emoji blocks,
# pausing with sleep between frames to simulate work being done.
progress() {
    sleep 1
    echo -ne ${b_green}'🟩🟩🟩🟩🟩    (33%)\r '      # print 33% bar, \r returns cursor to line start
    sleep 1
    echo -ne '🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩    (66%)\r '   # overwrite with 66% bar
    sleep 1
    echo -ne '🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩    (100%)\r ' # overwrite with 100% bar
    echo -ne "${nc}\n"   # reset color and move to new line
    sleep 2
    echo -e "${b_yellow}\n Êxito !\n${nc}"   # print "Success!" in Portuguese in bold yellow
}

# show_banner() — prints the large ASCII art "SOPA SPADES" title banner in green.
show_banner() {
    echo -e "${GREEN}"
    echo "  ███████╗ ██████╗ ██████╗  █████╗     ███████╗██████╗  █████╗ ██████╗ ███████╗███████╗"
    echo "  ██╔════╝██╔═══██╗██╔══██╗██╔══██╗    ██╔════╝██╔══██╗██╔══██╗██╔══██╗██╔════╝██╔════╝"
    echo "  ███████╗██║   ██║██████╔╝███████║    ███████╗██████╔╝███████║██║  ██║█████╗  ███████╗"
    echo "  ╚════██║██║   ██║██╔═══╝ ██╔══██║    ╚════██║██╔═══╝ ██╔══██║██║  ██║██╔══╝  ╚════██║"
    echo "  ███████║╚██████╔╝██║     ██║  ██║    ███████║██║     ██║  ██║██████╔╝███████╗███████║"
    echo "  ╚══════╝ ╚═════╝ ╚═╝     ╚═╝  ╚═╝    ╚══════╝╚═╝     ╚═╝  ╚═╝╚═════╝ ╚══════╝╚══════╝"
    echo -e "${nc}"
    echo "  Installer for Ubuntu 18+ / Debian / Pop!_OS / Zorin / Mint and derivatives"
    echo "  Source: https://github.com/atorresbr/sopaspades"
    echo ""
}

# -----------------------------------------------------------------------------
# MUST RUN AS ROOT
# -----------------------------------------------------------------------------
# whoami → prints the name of the current user.
# If it is not "root", show a trilingual error and exit with code 1.
if [ "$(whoami)" != "root" ]; then
    echo
    echo -e " 🔴 use the sudo command to run this script "
    sleep 1   # pause 1 second between each language message
    echo
    echo -e " 🔴 use el comando sudo para usar el script "
    sleep 1
    echo
    echo -e " 🔴 use o comando sudo para rodar o script de instalação "
    echo
    exit 1   # exit the script with a non-zero status to signal failure
fi

# MAIN SCRIPT STARTS HERE
# =============================================================================

   * * * =============================================== * * *
 *        🇪🇸   Executando los Comandos de Shell                *
   * * * =============================================== * * *
"

show_banner

echo -e "
   * * * =============================================== * * *
 *        🇧🇷   Executando os Comandos de Shell                 *

"

   * * * =============================================== * * *
 *        🇺🇸     Executing the Shell commands                  *
   * * * =============================================== * * *
"

show_banner
echo -ne "${b_green}"

echo -e "
   * * * =============================================== * * *
 *       🇪🇸 Este script instalará Sopa Spades en tu PC 🍜      *
   * * * =============================================== * * *
"
sleep 2
clear

show_banner
echo -ne "${b_green}"

echo -e "
   * * * =============================================== * * *
 *     🇺🇸 This script will install Sopa Spades on your PC 🍜   *
   * * * =============================================== * * *
"
sleep 2
clear

show_banner
echo -ne "${b_green}"

echo -e "
   * * * =============================================== * * *
 *   🇧🇷 Este script irá instalar o Sopa Spades em seu PC 🍜    *
   * * * =============================================== * * *
"
echo -e "${nc}"

# =============================================================================
# STEP 0 — CHECK SYSTEM COMPATIBILITY
# =============================================================================
step "STEP 0 — 
| ✨🇪🇸 Comprobando compatibilidad 
| ✨🇺🇸 Checking system compatibility 
| ✨🇧🇷 Verificando compatibilidade"

# lsb_release -si → prints the distributor ID (e.g. Ubuntu, Debian, Mint)
# lsb_release -sr → prints the release number (e.g. 22.04)
# 2>/dev/null → silences any error output if lsb_release is not installed
# || echo "Unknown" → fallback value if the command fails
DISTRO=$(lsb_release -si 2>/dev/null || echo "Unknown")
VERSION=$(lsb_release -sr 2>/dev/null || echo "Unknown")
info "Detected OS: $DISTRO $VERSION"

# grep -qiE → quiet (-q), case-insensitive (-i), extended regex (-E) search
# Checks /etc/os-release for lines matching ID= or ID_LIKE= containing "ubuntu" or "debian"
# This covers all distros derived from Debian/Ubuntu (Mint, Pop!_OS, Zorin, etc.)
if grep -qiE '^(ID|ID_LIKE)=.*\b(ubuntu|debian)\b' /etc/os-release 2>/dev/null; then
    ok "Debian-family OS detected ✅ — fully supported."
# command -v apt-get → checks if apt-get is available in PATH without running it
elif command -v apt-get &>/dev/null; then
    info "Unknown distro but apt-get found — proceeding..."
else
    die "This script requires a Debian/Ubuntu-based system (apt-get not found)."
fi

ok "Running as: $(whoami)"

# =============================================================================
# STEP 1 — REMOVE OLD INSTALLATIONS
# =============================================================================

| ✨🇪🇸 Eliminando el directorio antiguo
| ✨🇺🇸 Removing the old directory 
| ✨🇧🇷 Removendo o antigo diretório ✅"

sleep 2
info "🇧🇷 Removendo o antigo diretório para reinstalar o Jogo ✅"
sleep 2
echo
echo
echo -ne ${b_yellow}
sleep 2
echo -e " sudo rm -rf /usr/local/share/games/openspades "
sleep 2
echo
echo -e " && "
echo
sleep 2
echo -e " sudo rm -rf /usr/local/share/games/sopaspades "
echo -ne ${nc}
echo

rm -rf /usr/local/games/openspades   2>/dev/null || true   # remove old OpenSpades install; ignore errors if path doesn't exist
rm -rf /usr/local/games/sopaspades   2>/dev/null || true   # remove old SopaSpades install
rm -rf /usr/local/share/games/openspades 2>/dev/null || true  # alternate share location for OpenSpades
rm -rf /usr/local/share/games/sopaspades 2>/dev/null || true  # alternate share location for SopaSpades
rm -rf /usr/share/applications/openspades.desktop 2>/dev/null || true  # remove OpenSpades desktop launcher entry
rm -rf /usr/share/applications/sopaspades.desktop 2>/dev/null || true  # remove SopaSpades desktop launcher entry
rm -rf /usr/share/pixmaps/openspades.xpm 2>/dev/null || true  # remove OpenSpades icon from pixmaps
rm -rf /usr/share/pixmaps/sopaspades.xpm 2>/dev/null || true  # remove SopaSpades icon from pixmaps
rm -rf /usr/games/openspades 2>/dev/null || true   # remove binary from /usr/games
rm -rf /usr/games/sopaspades 2>/dev/null || true   # remove SopaSpades binary from /usr/games
rm -rf /home/*/.cache/icon-cache.kcache 2>/dev/null || true  # clear KDE icon cache for all users
rm -rf /home/*/.cache/thumbnails/*      2>/dev/null || true  # clear thumbnail cache for all users
rm -rf /home/*/.cache/icons/*           2>/dev/null || true  # clear icon cache directories for all users

# dpkg -l → list installed packages
# grep -q openspades → silently check if openspades appears in the list
if dpkg -l 2>/dev/null | grep -q openspades; then
    info "OPENSPADES was installed, removing it 🗑️"
    apt purge openspades -y   # apt purge → remove package + its config files; -y auto-confirms
else
    info "OpenSpades or SynSpades are not here 😬"
fi

echo
echo
progress

# =============================================================================
# STEP 2 — INSTALL DEPENDENCIES
# =============================================================================
step "STEP 2 — 
| ✨🇪🇸 Instalando las dependencias 
| ✨🇺🇸 Installing the dependencies 
| ✨🇧🇷 Instalando as dependências"

echo -ne ${b_green}
sleep 2
echo -e " 🇪🇸 Instalando las dependencias "
echo
sleep 2
echo -e " 🇺🇸 Installing the dependencies "
echo
sleep 2
echo -e " 🇧🇷 Instalando as dependências "
echo
echo
sleep 1
echo " sudo apt-get install "
echo -ne ${b_yellow}

info "🍺 pkg-config libglew-dev"
sleep 1
info "🍺 libcurl4-openssl-dev"
sleep 1
info "🍺 libsdl2-dev"
sleep 1
info "🍺 libsdl2-image-dev"
sleep 1
info "😋 libalut-dev"
sleep 1
info "🍺 xdg-utils libfreetype6-dev"
sleep 1
info "🍺 libopus-dev"
sleep 1
info "🍺 libopusfile-dev"
sleep 1
info "🍺 cmake imagemagick"
sleep 1
info "🍺 libjpeg-dev"
sleep 1
info "😋 🙏 libxinerama-dev"
sleep 1
info "🙏 libxft-dev"
sleep 1
info "✨😋🍺🍻 Uma ceva trincando é uma delícia !"
sleep 1
info "✨😋🍺🍻 some beer are so Belicious"
sleep 1
info "✨😋🍺🍻 Una cerbeza mui geladita és una Belícia ! Que rica !"

sleep 1

echo
echo
sleep 1

echo -ne ${b_green}
echo -ne '🟩🟩🟩🟩🟩    (33%)\r '
sleep 1
echo -ne '🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩    (66%)\r '
sleep 1
echo -ne '🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩    (100%)\r '
echo -ne "\n"

sleep 2
echo -e "${b_yellow}\n Êxito !\n${nc}"
sleep 1

echo -ne ${nc}

info "This may take several minutes on a slow connection... ☕"

apt-get install -y \
    build-essential \
    pkg-config \
    libglew-dev \
    libcurl4-openssl-dev \
    libsdl2-dev \
    libsdl2-image-dev \
    libalut-dev \
    xdg-utils \
    libfreetype6-dev \
    libopus-dev \
    libopusfile-dev \
    cmake \
    imagemagick \
    zip unzip \
    libjpeg-dev \
    libxinerama-dev \
    libxft-dev \
    libopenal-dev \
    zlib1g-dev

ok "All dependencies installed."

for pkg in libsdl2-dev libglew-dev libopus-dev libfreetype6-dev; do
    dpkg -s "$pkg" &>/dev/null && ok "$pkg installed" || die "$pkg failed to install."
done

progress

# =============================================================================
# STEP 3 — CLONE REPOSITORY
# =============================================================================
step "STEP 3 — 
| ✨🇪🇸 Clonando el repositorio 
| ✨🇺🇸 Cloning the repository 
| ✨🇧🇷 Clonando o repositório"

info "🇪🇸 Clonando el repositorio y instalando"
sleep 2
info "🇺🇸 Cloning the repository and installing"
sleep 2
info "🇧🇷 Clonando o repositório e instalando"
sleep 2
sleep 2

echo -ne ${b_yellow}
echo -e " sudo rm -Rf sopaspades && sudo rm -Rf openspades "
sleep 1
echo
echo -e " && "
echo
sleep 1
echo -e " git clone https://github.com/atorresbr/sopaspades.git && cd sopaspades "
echo
echo

sleep 1
echo -ne ${b_green}
echo -ne '🟩🟩🟩🟩🟩    (33%)\r '
sleep 1
echo -ne '🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩    (66%)\r '
sleep 1
echo -ne '🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩    (100%)\r '
echo -ne "\n"
sleep 2
echo -e "${b_yellow}\n Êxito !\n${nc}"
sleep 2
echo -ne ${nc}

ACTUAL_USER="${SUDO_USER:-$(logname 2>/dev/null || echo root)}"
USER_HOME=$(getent passwd "$ACTUAL_USER" | cut -d: -f6)
REPO_DIR="$USER_HOME/sopaspades"

# Remove old clone
rm -rf "$REPO_DIR" 2>/dev/null || true
rm -rf "$USER_HOME/openspades" 2>/dev/null || true
rm -rf "$USER_HOME/a-la-popa" 2>/dev/null || true

runuser -u "$ACTUAL_USER" -- git clone https://github.com/atorresbr/sopaspades.git "$REPO_DIR"

[ -f "$REPO_DIR/CMakeLists.txt" ] || die "CMakeLists.txt not found — clone may have failed."
ok "Repository cloned to $REPO_DIR"

progress

sleep 2

# =============================================================================
# STEP 4 — BUILD AND INSTALL
# =============================================================================
step "STEP 4 — 
| ✨🇪🇸 instalando 
| ✨🇺🇸 installing 
| ✨🇧🇷 instalando"

echo -ne ${b_green}
sleep 2
echo -e " 🇪🇸 instalando "
echo
sleep 2
echo -e " 🇺🇸 installing "
echo
sleep 2
echo -e " 🇧🇷 instalando "
echo
echo -ne ${nc}

info "mkdir sopaspades.mk && cd sopaspades.mk"
info "cmake .. -DCMAKE_BUILD_TYPE=RelWithDebInfo"
info "make -j\$(nproc) && sudo make install"

echo -ne ${b_yellow}
echo -e " mkdir sopaspades.mk && cd sopaspades.mk && "
sleep 1
echo -e " cmake .. -DCMAKE_BUILD_TYPE=RelWithDebInfo && "
sleep 1
echo -e " make && sudo make install "
sleep 1

echo
echo
sleep 1
echo -ne ${b_green}
echo -ne '🟩🟩🟩🟩🟩    (33%)\r '
sleep 1
echo -ne '🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩    (66%)\r '
sleep 1
echo -ne '🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩🟩    (100%)\r '
echo -ne "\n"
sleep 2
echo -e "${b_yellow}\n Êxito !\n${nc}"
sleep 1
echo -ne ${nc}

BUILD_DIR="$REPO_DIR/sopaspades.mk"
runuser -u "$ACTUAL_USER" -- mkdir -p "$BUILD_DIR"

cd "$BUILD_DIR"
runuser -u "$ACTUAL_USER" -- cmake .. -DCMAKE_BUILD_TYPE=RelWithDebInfo

ok "CMake configuration successful."
info "Build files written to: $BUILD_DIR"

CORES=$(( $(nproc) > 1 ? $(nproc) - 1 : 1 ))
info "Compiling with $CORES of $(nproc) cores (1 kept free) — please wait ☕ ..."
runuser -u "$ACTUAL_USER" -- make -j"$CORES"

[ -f "$BUILD_DIR/bin/sopaspades" ] || die "Binary not found after compilation. Check errors above."

ok "Compilation complete!"
ok "Binary: $BUILD_DIR/bin/sopaspades"

# =============================================================================
# STEP 4b — INSTALL SYSTEM-WIDE
# =============================================================================
step "STEP 4b — 
| ✨🇪🇸 Instalando sistema 
| ✨🇺🇸 Installing system-wide 
| ✨🇧🇷 Instalando no sistema"

info "Running sudo make install..."
make install

if command -v sopaspades &>/dev/null; then
    ok "sopaspades is now available system-wide."
else
    info "Binary not found in PATH yet. You can still run it directly:"
    info "  $BUILD_DIR/bin/sopaspades"
fi

progress

# =============================================================================
# STEP 5 — INSTALL MODERN PACK
# =============================================================================
MODERN_PACK="$REPO_DIR/MODERN-PACK/modern_pack.zip"
RESOURCES_DIR="$USER_HOME/.local/share/sopaspades/Resources"

if [ -f "$MODERN_PACK" ]; then
    step "BONUS — 
    | ✨🇪🇸 Instalando El MODERN-MODERN_PACK
    | ✨🇺🇸 Installing the MODERN-PACK skins 
    | ✨🇧🇷 Instalando O MODERN-PACK skins"
    info "Creating resources directory: $RESOURCES_DIR"
    runuser -u "$ACTUAL_USER" -- mkdir -p "$RESOURCES_DIR"
    info "Extracting modern_pack.zip..."
    runuser -u "$ACTUAL_USER" -- unzip -o "$MODERN_PACK" -d "$RESOURCES_DIR"
    ok "MODERN-PACK installed to $RESOURCES_DIR"
else
    info "MODERN-PACK zip not found — skipping."
    info "You can install it later manually from the repo."
fi

# =============================================================================
# DONE
# =============================================================================
echo ""
echo -e "${GREEN}"
echo "  ╔═══════════════════════════════════════════════════════════════════════╗"
echo "  ║  BUILD COMPLETE / INSTALAÇÃO COMPLETINHA! — Bom jogo! 🍜Sopa! 🔫🇧🇷    ║"
echo "  ╚═══════════════════════════════════════════════════════════════════════╝"
echo -e "${nc}"
echo "  To start the game, run:"
echo ""
echo "      sopaspades"
echo ""
echo "  Or run directly without installing:"
echo "      $BUILD_DIR/bin/sopaspades"
echo ""
echo "  If the game crashes on audio, install OpenAL:"
echo "      sudo apt-get install -y libopenal1"
echo ""
echo "  Source: https://github.com/atorresbr/sopaspades"
echo ""

echo -e " ✨😘🇪🇸 Comparte este script ! "
sleep 1
echo -e " ✨😘🇺🇸 Share this Script ! "
sleep 1
echo -e " ✨😘🇧🇷 Compartilhe este Script ! "
sleep 1

echo -e "${b_green} ✨🇪🇸 comenzando el Juego ${nc}"
sleep 1
echo -e "${b_green} ✨🇺🇸 starting the Game ${nc}"
sleep 1
echo -e "${b_green} ✨🇧🇷 iniciando o Jogo ${nc}"
sleep 1
echo

echo " 🔫🍜✨sopaspades ԅ(≖‿≖ԅ)  "
echo

if [ "$ACTUAL_USER" != "root" ]; then
    runuser -u "$ACTUAL_USER" -- /usr/games/sopaspades 2>/dev/null || \
    su - "$ACTUAL_USER" -c "/usr/games/sopaspades" 2>/dev/null || true
else
    echo "Run as regular user with: /usr/games/sopaspades"
fi

echo
