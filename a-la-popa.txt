# step() вҖ” clears the screen, redraws the banner, and prints a bold step header.
# Argument $1 is the step title text to display.
step() {

    echo ""
}

# ok() вҖ” prints a green [OK] success message.
ok() {
    echo -e "${GREEN}  [OK] $1${nc}"
}

# info() вҖ” prints a yellow [INFO] informational message.
info() {
    echo -e "${YELLOW}  [INFO] $1${nc}"
}

# die() вҖ” prints a red [ERROR] message and exits the script with code 1.
die() {
    echo -e "${RED}  [ERROR] $1${nc}"
    exit 1
}
